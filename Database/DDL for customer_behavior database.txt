CREATE DATABASE customer_behavior;
USE customer_behavior;

CREATE TABLE event_types (
    event_type_id INT IDENTITY(1,1) PRIMARY KEY,
    name VARCHAR(100) UNIQUE NOT NULL,
    description TEXT,
    created_at DATETIME DEFAULT GETDATE()
);

CREATE TABLE categories (
    category_id INT IDENTITY(1,1) PRIMARY KEY,
    name VARCHAR(150) UNIQUE NOT NULL,
    created_at DATETIME DEFAULT GETDATE()
);

CREATE TABLE payment_statuses (
    status_id INT IDENTITY(1,1) PRIMARY KEY,
    name VARCHAR(50) UNIQUE NOT NULL,
    created_at DATETIME DEFAULT GETDATE()
);

CREATE TABLE payment_methods (
    payment_method_id INT IDENTITY(1,1) PRIMARY KEY,
    name VARCHAR(100) UNIQUE NOT NULL,
    description TEXT,
    created_at DATETIME DEFAULT GETDATE()
);

CREATE TABLE locations (
    location_id INT IDENTITY(1,1) PRIMARY KEY,
    country_code VARCHAR(10),
    state VARCHAR(100),
    city VARCHAR(100),
    postal_code VARCHAR(20),
    latitude DECIMAL(10,6),
    longitude DECIMAL(10,6)
);

CREATE TABLE users (
    user_id INT IDENTITY(1,1) PRIMARY KEY,
    first_name VARCHAR(100),
    last_name VARCHAR(100),
    email VARCHAR(150) UNIQUE,
    phone VARCHAR(50),
    password_hash TEXT,
    is_active BIT DEFAULT 1,
    is_deleted BIT DEFAULT 0,
    created_at DATETIME DEFAULT GETDATE(),
    updated_at DATETIME DEFAULT GETDATE()
);

CREATE TABLE sessions (
    session_id INT IDENTITY(1,1) PRIMARY KEY,
    user_id INT,
    guest_id UNIQUEIDENTIFIER,

    start_time DATETIME,
    end_time DATETIME,

    device_type VARCHAR(50),
    user_agent TEXT,

    -- SQL Server uses VARCHAR for IP
    ip_address VARCHAR(45),

    source VARCHAR(100),
    campaign VARCHAR(100),

    created_at DATETIME DEFAULT GETDATE(),

    CONSTRAINT fk_sessions_user
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);

CREATE TABLE products (
    product_id INT IDENTITY(1,1) PRIMARY KEY,
    product_name VARCHAR(200),
    category_id INT,
    location_id INT,
    stock_quantity INT,
    avg_rating DECIMAL(3,2),
    total_reviews INT,
    created_at DATETIME DEFAULT GETDATE(),
    updated_at DATETIME DEFAULT GETDATE(),

    FOREIGN KEY (category_id) REFERENCES categories(category_id),
    FOREIGN KEY (location_id) REFERENCES locations(location_id)
);

CREATE TABLE product_prices (
    product_price_id INT IDENTITY(1,1) PRIMARY KEY,
    product_id INT,
    price DECIMAL(10,2) NOT NULL,
    start_date DATETIME,
    end_date DATETIME,
    created_at DATETIME DEFAULT GETDATE(),

    FOREIGN KEY (product_id) REFERENCES products(product_id)
);

CREATE TABLE cart_sessions (
    cart_id INT IDENTITY(1,1) PRIMARY KEY,
    session_id INT,
    created_at DATETIME DEFAULT GETDATE(),
    is_abandoned BIT DEFAULT 0,

    FOREIGN KEY (session_id) REFERENCES sessions(session_id)
);

CREATE TABLE cart_items (
    cart_item_id INT IDENTITY(1,1) PRIMARY KEY,
    cart_id INT,
    product_id INT,
    quantity INT,
    unit_price DECIMAL(10,2),
    total_price DECIMAL(10,2),
    updated_at DATETIME DEFAULT GETDATE(),

    FOREIGN KEY (cart_id) REFERENCES cart_sessions(cart_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
);

CREATE TABLE transactions (
    transaction_id INT IDENTITY(1,1) PRIMARY KEY,
    user_id INT,
    payment_method_id INT,
    status_id INT,
    total_amount DECIMAL(10,2),
    transaction_date DATETIME,
    location_id INT,
    created_at DATETIME DEFAULT GETDATE(),
    updated_at DATETIME DEFAULT GETDATE(),

    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (payment_method_id) REFERENCES payment_methods(payment_method_id),
    FOREIGN KEY (status_id) REFERENCES payment_statuses(status_id),
    FOREIGN KEY (location_id) REFERENCES locations(location_id)
);

CREATE TABLE transaction_items (
    transaction_item_id INT IDENTITY(1,1) PRIMARY KEY,
    transaction_id INT,
    product_id INT,
    quantity INT,
    unit_price DECIMAL(10,2),
    total_line DECIMAL(10,2),

    FOREIGN KEY (transaction_id) REFERENCES transactions(transaction_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
);

CREATE TABLE events (
    event_id INT IDENTITY(1,1) PRIMARY KEY,

    user_id INT,
    session_id INT,
    event_type_id INT,

    -- denormalized
    event_name VARCHAR(100),

    product_id INT,

    -- SQL Server JSON stored as NVARCHAR
    event_properties NVARCHAR(MAX),

    event_timestamp DATETIME DEFAULT GETDATE(),
    created_at DATETIME DEFAULT GETDATE(),

    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (session_id) REFERENCES sessions(session_id),
    FOREIGN KEY (event_type_id) REFERENCES event_types(event_type_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
);

CREATE TABLE reviews (
    review_id INT IDENTITY(1,1) PRIMARY KEY,
    user_id INT,
    product_id INT,
    score INT,
    review_text TEXT,
    is_verified BIT DEFAULT 0,
    likes_count INT DEFAULT 0,
    created_at DATETIME DEFAULT GETDATE(),
    updated_at DATETIME DEFAULT GETDATE(),

    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
);

CREATE INDEX idx_events_name ON events(event_name);
CREATE INDEX idx_events_user_time ON events(user_id, event_timestamp);
CREATE INDEX idx_transactions_user ON transactions(user_id);
CREATE INDEX idx_products_category ON products(category_id);
